#h5bp-on-flask

html5 boilerplate on flask

### Windows

    cd h5bp-on-flask

    virtualenv venv
    
    venv\scripts\activate
    
    pip install -r requirements.txt
    
    python runserver.py
    
    
### Mac/Linux
    
    cd h5bp-on-flask

    virtualenv --no-site-packages .
    
    . bin/activate
    
    bin/pip install -r requirements.txt
    
    python runserver.py
